## 예제 코드

        <CodeBlock>{`import { useQuery } from "@tanstack/react-query";
function useUser(userId: string) {
  return useQuery({
    queryKey: ["user", userId],
    queryFn: () => fetch(/api/users/userId).then(r => r.json()),
    staleTime: 60_000,
  });
}`}</CodeBlock>
